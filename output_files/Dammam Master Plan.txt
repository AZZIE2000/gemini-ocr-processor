الـنـفـايـات الـصـنـاعـيـة غـيـر الـخـطـرة
1 . تـوفـيـر مـنـاطـق تـجـمـيـع مـحـددة بـعـلامـات لـلـنـفـايـات غـيـر الـخـطـرة ، مـع ضـمـان سـهـولـة الـوصـول الـيـهـا وتـوفـيـر الـمـسـاحـة الـكـافـيـة لـاسـتـيـعـاب الـكـمـيـات الـمـنـتـجـة .
2 . فـرز الـمـنـزلات عـن الـمـصـدر لإعـادة تـدويـرهـا وإعـادة اسـتـخـدامـهـا ، مـثـل نـفـايـات الـزجـاج والـورق والـكـرتـون والـنـفـايـات الـبـلاسـتـيـكـيـة ، وأي الـمـكـونـات الـصـلـبـة الأخـرى فـي الـمـعـالـجـة والـتـخـلـص مـنـهـا .
3 . الـتـأكـد مـن أن كـل نـوع فـرز مـن الـنـفـايـات الـقـابـلـة لإعـادة الـتـدويـر تـجـمـع فـي الـحـاويـات الـخـاصـة بـهـا .
4 . مـعـالـجـة الـنـفـايـات مـعـالـجـة أولـيـة إذا لـزم الأمـر ، مـثـل الـتـمـزيـق أو الـتـفـكـيـك قـبـل نـقـلـهـا لـمـرا فـق الـمـعـالـجـة مـن أجـل إعـادة الاسـتـخـدام أو إعـادة الـتـدويـر أو حـتـى الـتـخـلـص الـنـهـائـي .
5 . الـتـأكـد مـن أن مـقـدمـي خـدمـات إدارة الـنـفـايـات الـمـرخـصـيـن فـقـط هـم مـن يـجـمـعـون الـنـفـايـات ويـنـقـلـونـهـا إلـى مـرا فـق الـمـعـالـجـة أو مـع أو مـقـع الـتـخـلـص الـمـرخـصـة

الـنـفـايـات الـصـنـاعـيـة الـخـطـرة

1 . تـوفـيـر مـنـاطـق جـمـع مـحـددة بـعـلامـات لـلـنـفـايـات الـصـنـاعـيـة الـخـطـرة ، بـحـيـث تـكـون مـسـاحـة كـافـيـة لـاسـتـيـعـاب الـكـمـيـات الـمـنـتـجـة وسـهـولـة الـوصـول .
2 . يـجـب أن تـكـون مـنـطـقـة الـجـمـع بـعـيـدة عـن الـمـنـاطـق الـسـكـنـيـة ، لـأن بـعـض الـنـفـايـات قـد تـنـقـلون الـحـاويـات الـسـائـلـة أو يـتـسـربـون فـي أ مـسـاكـهـا .
3 . الـتـأكـد مـن تـجـانـس الـنـفـايـات وتـنـظـيـمـهـا وحـسـب درجـة خـطـورتـهـا وخـصـائـصـهـا والـتـوصـيـات الـفـنـيـة .
ضـمـان جـمـع الـنـفـايـات وفـقـاً لـمـتـطـلـبـات صـيـانـة بـيـانـات الـسـلامـة .
4 . جـمـع الـنـفـايـات الـخـطـرة فـي حـاويـات أو جـاويـات مـصـنـوعـة مـن مـواد لا تـتـفـاعـل مـع الـنـفـايـات الـخـطـرة ويـجـب أن تـكـون سـلـيـمـة غـيـر قـادرة عـلـى تـغـيـر الـظـروف أو ضـرر مـمـكـن أن يـسـبـب تـسـرب لـلـمـواد الـمـخـزنـة ، وأن يـظـهـر عـلـى الـمـسـتـوعـبـات عـلامـات أو رقـعـات تـصـنـيـف الـخـطـورة .
5 . وجـوب إغـلاق الـحـاويـات دائـمـاً لـمـنـع الانـسـكـاب أو الـتـبـخـر ؛ كـمـا يـوصـى بـشـدة بـاسـتـخـدام أنـظـمـة احـتـواء ثـانـويـة .
6 . وجـوب تـركـيـب مـعـدات الـحـمـايـة مـن الـحـرائـق والإنـقـاذ فـي مـنـاطـق جـمـع الـنـفـايـات الـخـطـرة .
7 . الـتـأكـد مـن أن مـقـدمـي خـدمـات إدارة الـنـفـايـات الـمـعـتـمـديـن يـسـتـخـدمـون نـظـام " الـمـانـيـفـسـت الإلـكـتـرونـي " لـجـمـع ونـقـل الـنـفـايـات الـخـطـرة ، وكـذلـك الـمـنـتـجـيـن إذا قـامـوا بـتـخـزيـن الـنـفـايـات لـفـتـرات مـتـوسـطـة / طـويـلـة .
8 . الـحـرص عـلـى نـقـل الـنـفـايـات بـطـريـقـة سـلـيـمـة بـيـنـيـا ، مـن قـبـل شـركـات نـقـل والـنـفـايـات الـخـطـرة الـمـرخـصـة لـتـجـنـب الانـسـكـابـات الـعـرضـيـة ولـتـتـبـع الـنـقـل والـنـوعـيـة الـبـيـانـيـة .
